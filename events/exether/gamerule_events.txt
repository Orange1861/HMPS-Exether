namespace = gamerule

gamerule.1 = {
	type = country_event
	placement = root

	title = gamerule.1.t
	dlc = dlc_roe
	hidden = yes

	duration = 1
	immediate = {
		if = {
			limit = {
				has_game_rule = bloodtithe_mode_basic
			}
			every_country = {
				add_journal_entry = {
					type = je_vampire_university
				}
			}
		}

		if = {
			limit = {
				has_game_rule = bloodtithe_mode_player_basic
			}
			every_country = {
				limit = {
					is_ai = no
				}
				add_journal_entry = {
					type = je_vampire_university
				}
			}
		}

		if = {
			limit = {
				has_game_rule = bloodtithe_mode_crazy
			}
			every_country = {
				add_journal_entry = {
					type = je_vampire_university
				}

				add_journal_entry = {
					type = je_blood_tribunal
				}
			}
		}

		if = {
			limit = {
				has_game_rule = bloodtithe_mode_player_crazy
			}
			every_country = {
				limit = {
					NOT = { 
						any_primary_culture = {
							has_discrimination_trait = vampiric
						} 
					}
					is_ai = no
				}

				add_journal_entry = {
					type = je_vampire_university
				}

				add_journal_entry = {
					type = je_blood_tribunal
				}
			}
		}
	}
}
