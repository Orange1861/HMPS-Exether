buramog_war.1 = {
	type = country_event
	placement = scope:invasion_base_state

	title = buramog_war.1.t
	desc = buramog_war.1.d
	flavor = buramog_war.1.f

	dlc = dlc_roe

	event_image = {
		video = "gfx/event_pictures/unspecific_sick_in_hospital.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_military.dds"

	duration = 3

	cooldown = { days = short_modifier_time }

	trigger = {
		OR = {
			exists = c:BDB
			exists = c:RTD
		}
	}

	immediate = {
		root = {
			save_scope_as = invader_scope
		}

		random_primary_culture = {
			save_scope_as = invader_culture
		}

		random_scope_state = {
			limit = {
                any_neighbouring_state = {
					OR = {
						owner ?= c:RTD
						owner ?= c:BDB
					}
                   
                }
			}
			save_scope_as = invasion_base_state
			
		}	
		scope:invasion_base_state = {
			random_neighbouring_state = {
				limit = {
					OR = {
						c:RTD ?= owner
						c:BDB ?= owner
					}
				}
				owner = {
					save_scope_as = adversary_scope
				}
			}
		}
	}

	option = {
		name = buramog_war.1.a
		default_option = yes
		scope:invasion_base_state = {
			add_modifier = {
				name = modifier_failed_incursion 
				days = short_modifier_time
			}	
			this.state_region = {
				add_devastation = 15
			}
			add_radicals_in_state = {
				value = medium_radicals
				culture = cu:buramog
			}
		}
		scope:adversary_scope = {
			post_notification = plaguewood_invader_failed_attack_notification
		}
	}

	option = {
		name = buramog_war.1.b
		scope:invasion_base_state = {
			add_modifier = {
				name = modifier_failed_incursion 
				days = short_modifier_time
			}
			this.state_region = {
				add_devastation = 5
			}
			add_radicals_in_state = {
				value = medium_radicals
				pop_type = officers
				culture = scope:invader_culture
			}
			this.region = {
				random_scope_state = {
					limit = {
						owner = ROOT
						any_scope_building = {
							is_building_type = building_barracks
						}	
					}
					add_radicals_in_state = {
						value = small_radicals
						pop_type = officers
						culture = scope:invader_culture
					}
				}
			}	
		}
		ig:ig_armed_forces = {
			add_modifier = {
				name = modifier_failed_to_cover_incursion
				months = 13
			}
		}
		scope:adversary_scope = {
			post_notification = plaguewood_invader_failed_attack_notification
		}
	}
}

			}

	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	duration = 3

	cooldown = { days = very_long_modifier_time }

	trigger = {
		any_country = {
			has_journal_entry = je_buramog_raids
			NOT = {
				has_truce_with = ROOT
			}
		}
		NOT = {
			has_variable = buramog_council_var
		}
		NOT = {
			is_subject = yes
		}
	}

	immediate = {
		set_variable = buramog_council_var
		random_country = {
			limit = {
				has_journal_entry = je_buramog_raids
				NOT = {
					has_truce_with = ROOT
				}
			}
			save_scope_as = invader_scope
		}
		capital = {
			save_scope_as = capital_scope
		}

		set_global_variable = {
			name = plaguewood_attacker_var
			value = scope:invader_scope
		}
		set_global_variable = {
			name = plaguewood_defender_var
			value = ROOT
		}
	}

	option = {
		name = buramog_war.7.a
		default_option = yes
		if = {
			limit = {
				NOT = {
					has_law = law_type:law_professional_army
				}
			}
			activate_law = law_type:law_professional_army
		}
		add_modifier = {
			name = modifier_circassian_last_stand
			days = normal_modifier_time
		}
		create_character = {
			first_name = Hogmog
			last_name = Kvetz
			birth_date = 1139.6.1
			ruler = yes
			hq = region_plaguewood
			is_general = yes
			commander_rank = commander_rank_3
			interest_group = ig_landowners
			ideology = ideology_moderate
			traits = {
				persistent
				brave
				forest_commander
			}
		}
	}

	option = {
		name = buramog_war.7.b
		trigger = {
			any_country = {
				country_rank >= rank_value:major_power
				has_interest_marker_in_region = region_plaguewood
				NOT = {
					has_journal_entry = je_buramog_raids
				}
			}
		}
		create_character = {
			first_name = Hogmog
			last_name = Kvetz
			birth_date = 1139.6.1
			ruler = yes
			hq = region_plaguewood
			is_general = yes
			commander_rank = commander_rank_3
			interest_group = ig_landowners
			ideology = ideology_moderate
			traits = {
				persistent
				brave
				forest_commander
			}
		}
		add_modifier = {
			name = modifier_buramog_diplomacy
			days = normal_modifier_time
		}
		every_country = {
			limit = {
				country_rank >= rank_value:major_power
				NOT = {
					has_journal_entry = je_buramog_raids
				}
				OR = {
					has_interest_marker_in_region = region_plaguewood
					has_diplomatic_pact = {
						who = scope:invader_scope
						type = rivalry
					}	
				}
			}
			trigger_event = { id = buramog_war.8 }
		}
	}

	option = {
		name = buramog_war.7.c
		trigger = {
			OR = {
				AND = {
					exists = c:RCR
					c:RCR = {
                        OR = {
                            relations:root >= relations_threshold:cordial
                        }
                    }
				}
				AND = {
					exists = c:OCR
					c:OCR = {
                        OR = {
                            relations:root >= relations_threshold:cordial
                        }
                    }
				}
			}
		}
		if = {
			limit = {
				NOT = {
					country_has_primary_culture = cu:korkaz
				}
			}
			add_primary_culture = cu:korkaz
		}
		if = {
			limit = {
				c:OCR = {
                    relations:root >= relations_threshold:cordial
                }
				NOT = {
							c:OCR = {
								any_scope_treaty = {
									binds = root
									any_scope_article = {
										has_type = alliance
									}
								}
							}
				}
			}
			create_treaty = {
				first_country = root
				second_country = c:OCR

				is_draft = no
				binding_period = { years = 25  }

				articles_to_create = {
					{ 
						article = alliance
					}


				}
			}
		}
		else_if = {
			limit = {
				c:RCR = {
                    relations:root >= relations_threshold:cordial
                }
				NOT = {
							c:RCR = {
