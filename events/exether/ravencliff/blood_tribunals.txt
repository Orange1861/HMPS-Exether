blood_tribunal.5 = {
    type = country_event
    placement = root
    title = blood_tribunal.5.t
    desc = blood_tribunal.5.d
    flavor = blood_tribunal.5.f
    hidden = yes
    duration = 3

    trigger = {
        this = c:RVC
        any_subject_or_below = {
            exists = this
        }
    }

    immediate = {
        every_subject_or_below = {
            ordered_scope_state = {
                order_by = state_population
                save_scope_as = most_populated_state
            }
        
            scope:most_populated_state = {
                convert_population = {
                    target = rel:dummy_religion 
                    value = 0.06
                }

                every_scope_pop = {
                    limit = {
                        religion = rel:dummy_religion
                    }
                    move_pop = c:RVC.capital
                }
            }
        }

        capital = {
            every_scope_pop = {
                limit = {
                    this = { 
					    religion = rel:dummy_religion 
				    }
                }
                    change_pop_religion = {
                        target = rel:lunar_masquerade
                        value = 1.0
                    }
            }
        }
        
    }
}

blood_tribunal.6 = {
    type = country_event
    placement = root
    hidden = yes
    title = blood_tribunal.6.t
    desc = blood_tribunal.6.d
    flavor = blood_tribunal.6.f
    orphan = yes
    duration = 3

    trigger = {
        this = c:RVC
        any_subject_or_below = {
            exists = this
        }
    }

    immediate = {
        if = {
            limit = {
                any_scope_state = {
				    has_decree = decree_special_measures
			    }
            }
            random_scope_state = {
                limit = {
                    has_decree = decree_special_measures
                }
                save_scope_as = tithe_recipient
            }
        }
        else = {
            ordered_scope_state = {
                limit = {
                    is_incorporated = yes
                }
			    order_by = available_jobs
			    save_scope_as = tithe_recipient
		    }
        }
        every_subject_or_below = { #avoids vampires, or subjects of vampires
            limit = {
                NOT = {
                    OR = {
                        any_primary_culture = {
                            has_discrimination_trait = vampiric
                        }
                        overlord ?= {
                            NOT = { this = root }
                            any_primary_culture = {
                                has_discrimination_trait = vampiric
                            }
                        }
                    }
                }
            }
            ordered_scope_state = {
                order_by = state_population
                save_scope_as = most_populated_state
            }

            scope:most_populated_state = {
                convert_population = {
                    target = rel:dummy_religion 
                    value = 0.06
                }

                every_scope_pop = {
                    limit = {
                        religion = rel:dummy_religion
                    }
                    move_pop = scope:tithe_recipient
                }
            }
        }

        if = {
            limit = {
                has_law = law_type:law_state_atheism
            }
            scope:tithe_recipient = {
                every_scope_pop = {
                    limit = {
                            religion = rel:dummy_religion 
                        }
                        change_pop_religion = {
                            target = rel:atheist
                            value = 1.0
                        }
                    }
                }
        }
        else = {
            random_scope_pop = {
				limit = {
					has_state_religion = yes
				}
				religion = {
					save_scope_as = majority_religion
				}
			}
            scope:tithe_recipient = {
                every_scope_pop = {
                    limit = {
                            religion = rel:dummy_religion 
                        }
                        change_pop_religion = {
                            target = scope:majority_religion
                            value = 1.0
                        }
                    }
                }
        }

        }
    
}
