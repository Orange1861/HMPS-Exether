dragonblight.4 = { #hidden event
	type = country_event
	placement = root
	hidden = yes

	immediate = {
		every_scope_state = {
			limit = {
				is_split_state = yes
				state_is_in_dragon_isles = yes
				NOT = { has_variable = dragon_killswitch_var }
			}
			dragonblight_killswitch_engage = yes
			set_variable = dragon_killswitch_var
		}
		if = {
			limit = {
				NOT =  { has_variable = kobold_land_secession_lobby }
			}
			trigger_event = { id = dragonblight.5 popup = yes }
		}
	}

}

									}
								}
						}	
					}
					random_neighbouring_state = {
						limit = {
							owner = {
								OR = {
									country_has_primary_culture = cu:kotskarvi
										country_has_primary_culture = cu:lomarash
										country_has_primary_culture = cu:drogori
																			country_has_primary_culture = cu:garagoz
									country_has_primary_culture = cu:tskharvani

									country_has_primary_culture = cu:larltri
								}
							}
						}

						}
					}
					scope:outbreak_state = {
						dragonblight_killswitch_engage = yes
						set_state_owner = scope:lilybaeum_secession_scope

							limit = {

						}
					}
				}
				else_if={
					limit = {

					exists = c:MVI
					}
					scope:outbreak_state = {
						dragonblight_killswitch_engage = yes
						set_state_owner = c:MVI
					}
				}
				else_if={
					limit = {
					#Relevant kobold tag

					}

						dragonblight_killswitch_engage = yes
						set_state_owner = c:TKR
					}
				}	
				else = {
					scope:outbreak_state = {
						dragonblight_killswitch_engage = yes
					}
					create_country = {
						tag = MVI
						origin = this
						state = scope:outbreak_state.region_state:DIS

			}
		}	
	}
	option = {
