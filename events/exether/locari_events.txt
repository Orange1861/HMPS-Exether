    immediate = {
        s:STATE_ELENIKAAJ = {
            random_scope_state = {
                limit = {
                    any_scope_building = {

            }
        }
        set_variable = draconic_mural_variable
        add_modifier = {
            name = draconic_obelisk_murals
        }
    }

    option = {
        name = locari.1.a
        default_option = yes
        add_modifier = {
            name = draconic_carvings_examined
            days = short_modifier_time
        }
        add_journal_entry = {
            type = je_dragon_tombs
        }
        s:STATE_ELENIKAAJ = {
            random_scope_state = {
                add_modifier = {
                    name = modifier_dragon_tomb_site
                }
            }
        }
        s:STATE_TSURWA = {
            random_scope_state = {
                add_modifier = {
                    name = modifier_dragon_tomb_site
                }
            }
        }
        s:STATE_BORKAGAUW = {
            random_scope_state = {
                add_modifier = {
                    name = modifier_dragon_tomb_site
                }
            }
        }
        s:STATE_VAROTSOA = {
            random_scope_state = {
                add_modifier = {
                    name = modifier_dragon_tomb_site
                }
            }
        }
        s:STATE_ONOLYSTU = {
            random_scope_state = {
                add_modifier = {
                    name = modifier_dragon_tomb_site
                }
            }
        }
        custom_tooltip = locari_obelisk_tt
    }
}

locari.103 = { #notification event for OLC and EME
    type = country_event
    dlc = dlc_roe
    title = locari.103.t
    desc = locari.103.d
    flavor = {
        first_valid = {
			triggered_desc  = {
				desc = locari.103.f
				trigger = {
					root = {
                        c:EME ?= this
                    }
                    exists = c:OLC
				}
			}
            triggered_desc  = {
				desc = locari.103.f2
				trigger = {
					root = {
                        c:OLC ?= this
                    }
				}
			}
            triggered_desc  = {
				desc = locari.103.f3
				trigger = {
					AND = {
                        root = {
                            c:EME ?= this
                        }
                        NOT = {
                            exists = c:OLC
                        }
                    }
				}
			}
        }
    }
    event_image = {
        video = "bear_talking_to_bureaucrats_positive"
    }
    
    on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

    icon = "gfx/interface/icons/event_icons/event_map.dds"
    
    duration = 3

    immediate = {
       c:OLC ?= {
            save_scope_as = olc_scope
       }
       
       c:ORL ?= {
            save_scope_as = orl_scope
       } 
    }
    option = { #we will see if they can eat it
        name = locari.103.a
        default_option = yes
        trigger = {
		    root = {
                c:EME ?= this
            }
            exists = c:OLC
		}
        show_as_tooltip = {
            c:OLC ?= {
                create_diplomatic_pact = {
			        country = c:ORL
			        type = puppet
		        }
            }  
        }    
    }
    option = { #that's cool
        name = locari.103.c
        default_option = yes
        trigger = {
            OR = {
                AND = {
                    root = {
                        c:EME ?= this
                    }
                    NOT = {
                        exists = c:OLC
                    }
                }
                NOT = {
                    root = {
                        c:EME ?= this
                    }
                }
            }
		}
        show_as_tooltip = {
            create_diplomatic_pact = {
			    country = c:ORL
			    type = puppet
		    }
        }    
    }

    option = { #NAW BRO
        name = locari.103.b
        trigger = {
			root = {
                c:EME ?= this
                }
            exists = c:OLC
			}
        c:EME ?= {
            create_diplomatic_pact = {
			    country = c:ORL
			    type = puppet
		    }
            create_diplomatic_catalyst = {

			    target = c:OLC
		    }
        }
