### COST ###
#######################
#######################

get_cost = {
	hidden_effect = {
		if = {
			limit = {
				NOT = { has_variable = num_arcana }
			}
			set_variable = {
				name = num_arcana
				value = 0
			}
		}
		
		change_variable = {
			name = num_arcana
			add = 1
		}
	
		if = {
			limit = {
				has_modifier = country_arcana_card_upkeep
			}
			remove_modifier = country_arcana_card_upkeep
		}
	
		add_modifier = {
			name = country_arcana_card_upkeep
			multiplier = var:num_arcana
		}
	}
}


### DISPEL ###
#######################
#######################

remove_cost = {
	hidden_effect = {
		change_variable = {
			name = num_arcana
			subtract = 1
		}
	
		if = {
			limit = {
				has_modifier = country_arcana_card_upkeep
			}
			remove_modifier = country_arcana_card_upkeep
		}
	
		if = {
			limit = {
				var:num_arcana > 0
			}
			add_modifier = {
				name = country_arcana_card_upkeep
				multiplier = var:num_arcana
			}
		}
	}
}
