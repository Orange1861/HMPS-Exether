STATE_RAHKIS_state_name_assign = {
    if = {
        limit = {
			owner ?= {
				any_primary_culture = { 
                    OR = {
                        AND = {
                            has_discrimination_trait = human_heritage
                            has_discrimination_trait = imperial_culture_group
                        }
                        AND = {
                            has_discrimination_trait = settler_coast_culture_group
                            has_discrimination_trait = human_heritage
                        }
                        AND = {
                            has_discrimination_trait = uriks_passage_culture_group
                            has_discrimination_trait = human_heritage
                            NOT = { cu:ocelian ?= this  }
                        }

                        
                    } 
                }
				NOT = {
                    any_primary_culture = { 
                        has_discrimination_trait = giant_heritage
                    } 
                }
			}
        }

        set_state_name = STATE_RAHKIS_settlers

    }
    else_if = {
        limit = {
			owner ?= {
                OR = {
                    country_has_primary_culture = cu:bakibi
                    any_primary_culture = { 
                        OR = {
                            AND = {
                                has_discrimination_trait = human_heritage
                                has_discrimination_trait = vaythir_culture_group
                            }
                            AND = {
                                has_discrimination_trait = uriks_passage_culture_group
                                has_discrimination_trait = human_heritage
                                NOT = { has_discrimination_trait = imperial_culture_group }
                            }
                            
                        } 
                    }
                }
				NOT = {
                    any_primary_culture = { 
                        has_discrimination_trait = giant_heritage
                    } 
                }
			}
        }

        set_state_name = STATE_RAHKIS_ocelian

    }

    else_if = {
        limit = {
            owner ?= {
                any_primary_culture = { 
                    OR = {
                        AND = {
                            has_discrimination_trait = dragon_heritage
                        }
                    } 
                }
                NOT = {
                    any_primary_culture = { 
                        has_discrimination_trait = giant_heritage
                    } 
                }
            }
        }

        set_state_name = STATE_RAHKIS_dragon

    }

    else_if = {
        limit = {
            owner ?= {
                any_primary_culture = { 
                    OR = {
                        AND = {
                            has_discrimination_trait = elf_heritage
                        }
                    } 
                }
                NOT = {
                    any_primary_culture = { 
                        has_discrimination_trait = giant_heritage
                    } 
                }
            }
        }

        set_state_name = STATE_RAHKIS_high_elf

    }

    else_if = {
        limit = {
            owner ?= {
                any_primary_culture = { 
                    OR = {
                        AND = {
                            has_discrimination_trait = bird_heritage
                        }
                    } 
                }
                NOT = {
                    any_primary_culture = { 
                        has_discrimination_trait = giant_heritage
                    } 
                }
            }
        }

        set_state_name = STATE_RAHKIS_bird

    }

    else = { # switch back to defaults
        reset_state_name = yes
        reset_hub_names = yes
    }
}
