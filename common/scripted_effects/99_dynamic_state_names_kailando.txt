STATE_CREUCHE_state_name_assign = {
    if = {
        limit = {
			owner ?= {
				any_primary_culture = { 
                    OR = {
                        AND = {
                            has_discrimination_trait = human_heritage
                            has_discrimination_trait = imperial_culture_group
                        }
                        AND = {
                            has_discrimination_trait = settler_coast_culture_group
                            has_discrimination_trait = human_heritage
                        }
                        AND = {
                            has_discrimination_trait = uriks_passage_culture_group
                            has_discrimination_trait = human_heritage
                            NOT = { cu:ocelian ?= this  }
                        }

                        
                    } 
                }
				NOT = {
                    any_primary_culture = { 
                        OR = {
                            cu:east_lacertian ?= this 
                            cu:kailando_lacertian ?= this
                            cu:west_lacertian ?= this
                        } 
                    } 
                }
			}
        }

        set_state_name = STATE_CREUCHE_settlers_coast

        set_hub_name = {
            type = city
            name = HUB_NAME_STATE_CREUCHE_city_settlers_coast
        }

		set_hub_name = {
            type = farm
            name = HUB_NAME_STATE_CREUCHE_farm_settlers_coast
        }

    }
    
    else = { # switch back to defaults
        reset_state_name = yes
        reset_hub_names = yes
    }
}

                            NOT = { cu:ocelian ?= this }
                        }
                        
                    } 
                }
                NOT = {
                    any_primary_culture = { 
                        OR = {
                            cu:east_lacertian ?= this 
                            cu:kailando_lacertian ?= this
                            cu:west_lacertian ?= this
                        } 
                    } 
                }
            }
        }

        set_state_name = STATE_NOLOGO_RULL_settlers_coast

        set_hub_name = {
            type = city
            name = HUB_NAME_STATE_NOLOGO_RULL_city_settlers_coast
        }

        set_hub_name = {
            type = port
            name = HUB_NAME_STATE_NOLOGO_RULL_port_settlers_coast
        }

    }
    else_if = {
        limit = {
            owner ?= {
                OR = {
                    country_has_primary_culture = cu:bakibi
                    any_primary_culture = { 
                        OR = {
                            AND = {
                                has_discrimination_trait = human_heritage
                                has_discrimination_trait = vaythir_culture_group
                            }
                            AND = {
                                has_discrimination_trait = uriks_passage_culture_group
                                has_discrimination_trait = human_heritage
                                NOT = { has_discrimination_trait = imperial_culture_group }
                            }
                            
                        } 
                    }
                }
                NOT = {
                    any_primary_culture = { 
                        OR = {
                            cu:east_lacertian ?= this 
                            cu:kailando_lacertian ?= this
                            cu:west_lacertian ?= this   
                        } 
                    } 
                }
            }
        }

        set_state_name = STATE_NOLOGO_RULL_ocelian

        set_hub_name = {
            type = port
            name = HUB_NAME_STATE_NOLOGO_RULL_port_ocelian
        }


    }
    else = { # switch back to defaults
        reset_state_name = yes
        reset_hub_names = yes
    }
