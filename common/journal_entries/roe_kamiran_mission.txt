je_kamiran_missions = {
	icon = "gfx/interface/icons/event_icons/event_religion.dds"

	group = je_group_historical_content

	scripted_button = DUM_deport_missionaries_button

	is_shown_when_inactive = {
		exists = c:DUM
		THIS = c:DUM
	}

	possible = {
		any_neighbouring_state = {
			owner = {
				has_law = law_type:law_theocracy
				country_has_state_religion = rel:cult_of_the_sun
			}
		}
		NOT = {
			has_law = law_type:law_isolationism
		}
	}

	immediate = {
		custom_tooltip = {
			text = kamiran_missionary_activities_begin_tt
			trigger_event = {
				id = kamiran.1
			}
		}
		every_scope_state = {
			limit = {
				any_neighbouring_state = {
					owner = {
						has_law = law_type:law_theocracy
						country_has_state_religion = rel:cult_of_the_sun
					}
				}
			}
			set_variable = open_to_missionaries_state
		}
	}

	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					has_law = law_type:law_no_migration_controls
					any_scope_state = {
						NOT = {
							has_variable = open_to_missionaries_state
						}
					}
				}
				every_scope_state = {
					limit = {
						NOT = {
							has_variable = open_to_missionaries_state
						}
					}
					set_variable = open_to_missionaries_state
				}
			}
			else = {
				every_scope_state = {
					limit = {
						any_neighbouring_state = {
							owner = {
								has_law = law_type:law_theocracy
								country_has_state_religion = rel:cult_of_the_sun
							}
						}
						NOT = {
							has_variable = open_to_missionaries_state
						}
					}
					set_variable = open_to_missionaries_state
				}
			}
		}
		random_events = {
			500 = 0
			10 = DUM_missionaries.1
			10 = DUM_missionaries.2
			10 = DUM_missionaries.3
			10 = DUM_missionaries.4
			10 = DUM_missionaries.5
		}
	}

	complete = {
		OR = {
			country_has_state_religion = rel:cult_of_the_sun
		}
	}

	on_complete = {
		if = {
			limit = {
				exists = c:AUM
				c:AUM = {
					is_subject_of = ROOT
					is_player = no
				}
			}
			annex = c:AUM
		}
		add_modifier = {
			name = colonial_zeal
			months = 60
		}
	}

	fail = {
		custom_tooltip = {
			text = dum_missionary_ban_tt
			has_variable = DUM_deported_missionaries
		}
		OR = {
			has_law = law_type:law_state_religion
			has_law = law_type:law_state_atheism
			has_law = law_type:law_isolationism
			has_law = law_type:law_closed_borders
		}
	}

	on_fail = {
		if = {
			limit = {
				exists = c:AUM
				c:AUM = {
					is_subject_of = ROOT
				}
			}
			if = {
				limit = {
					c:AUM = {
						is_subject_type = subject_type_protectorate
					}
				}
				remove_diplomatic_pact = {
					country =  c:AUM
					type = protectorate
				}
			}
			else_if = {
				limit = {
					c:AUM = {
						is_subject_type = subject_type_puppet
					}
				}
				remove_diplomatic_pact = {
					country =  c:AUM
					type = puppet
				}
			}


		}

		

		add_modifier = {
			name = paragon_of_unity
			months = 60
		}
		ig:ig_devout = {
		add_modifier = {
			name = banished_kamiran_heretics
			months = 60
		}
		}
	}

	weight = 10000
	should_be_pinned_by_default = yes
}

je_kamiran_takeover = {
	icon = "gfx/interface/icons/event_icons/event_protest.dds"

	group = je_group_historical_content

	is_shown_when_inactive = {
		exists = c:DUM
		THIS = c:DUM
		has_journal_entry = je_kamiran_missions
	}

	possible = {
		custom_tooltip = {
			text = missionaries_enough_time_tt
			OR = {
				AND = {
					has_variable = missionary_works_counter
					var:missionary_works_counter >= 5
					game_date >= 1188
				}
				game_date >= 1198
			}
		}
		is_subject = no
		any_scope_state = {
			is_incorporated = yes
			any_scope_pop = {
				has_pop_religion = cult_of_the_sun
			}
			is_capital = no
			state_region = { 
				OR = {
					is_homeland = cu:dumonian
					is_homeland = cu:ralaurian
					is_homeland = cu:dewaric
					is_homeland = cu:gelidrumin
					is_homeland = cu:avalorean
					is_homeland = cu:uplander
				}
			}
			turmoil > 0.15
		}
	}

	immediate = {
		random_scope_state = {
			limit = {
				is_incorporated = yes
				any_scope_pop = {
					has_pop_religion = cult_of_the_sun
				}
				is_capital = no
				state_region = { 
				OR = {
					is_homeland = cu:dumonian
					is_homeland = cu:ralaurian
					is_homeland = cu:dewaric
					is_homeland = cu:gelidrumin
					is_homeland = cu:avalorean
					is_homeland = cu:uplander
				}
				}
				turmoil > 0.15
			}
			add_modifier = {
				name = kamiran_cult
				days = short_modifier_time
			}
		}
		trigger_event = {
			id = kamiran.2
		}
	}

	on_monthly_pulse = {
		random_events = {
			5 = 0
			1 = kamiran.3
		}
	}

	complete = {
		OR = {
		NOT = {
			any_scope_state = {
				has_modifier = kamiran_cult
			}
		}
		country_has_state_religion = rel:cult_of_the_sun
		}

	}

	fail = {
		any_scope_state = {
			has_modifier = kamiran_cult
			count >= 4
		}
		any_scope_state = {
			AND = {
				has_modifier = kamiran_cult
				hidden_trigger = {
					any_scope_pop = {
						has_pop_religion = cult_of_the_sun
					}
				}
				turmoil >= 0.20
			}
		}
	}

	on_fail = {
		trigger_event = {
			id = kamiran.4
		}
	}

	weight = 10000
	should_be_pinned_by_default = yes
}
