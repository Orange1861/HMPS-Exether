has_arcana_gryphon = {
	type = country
	random_valid = no

	text = {
		trigger = {
			religion = {
				has_discrimination_trait = gryphon_domain
			}
		}
		localization_key = custom_loc_gryphon_heading
	}

	text = {
		trigger = {
			religion = {
				NOT = { has_discrimination_trait = gryphon_domain }
			}
		}
		localization_key = custom_loc_gryphon_heading_not
	}
}
