raid = {
    is_shown = {
		root = {country_has_primary_culture = cu:south_sea_ravager}
	}

	possible = {
		NOT = { has_modifier = raid_cooldown}
		NOT = { root.navy_size < 10}
	}
	
	when_taken = {
		add_modifier = {
			name = raid_cooldown
			months = 12
		}
		trigger_event = {
			id = pirate_raid.1
			popup = yes
		}
	}
	
	ai_chance = {
		value = 1
	}
}
