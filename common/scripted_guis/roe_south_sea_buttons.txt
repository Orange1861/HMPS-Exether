	south_sea_send_gift = {
		effect = {

			add_treasury = {
				add = 50
				divide = 1000
				multiply = root.gdp
				multiply = -1
			}

			custom_tooltip = {
				text = south_sea_favor_add_100_tt
			}

			hidden_effect = {
				set_variable = {
					name = south_sea_send_gift_cooldown
					value = yes
					days = 365
				}		
				change_variable = {
					name = south_sea_favor_var
					add = 100
				}
			}
	
		}
		ai_chance = {
			base = 0
			modifier = {
				trigger = { 
					weekly_net_fixed_income > money_amount_multiplier_medium
					gold_reserves > 0
	
				}
				add = 1
			}
			modifier = {
				trigger = {
					has_strategy = ai_strategy_armed_isolationism
				}
				add = -1000
			}		
			modifier = {
				OR = {
					is_diplomatic_play_committed_participant = yes
					is_at_war = yes
				}
				add = -1000
			}		
		}
		is_valid = {
			scaled_debt <= 0

			custom_tooltip = {
				text = has_south_sea_send_gift_cooldown_tt
				NOT = {
					has_variable = south_sea_send_gift_cooldown
				}
			}
		}
	}
